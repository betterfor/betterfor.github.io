package main

import (
   "encoding/json"
   "fmt"
   "time"
)

const textJson = `{"name":"xiaoming","duration":"5s"}`

func main() {
   var o Object
   json.Unmarshal([]byte(textJson),&o)
   fmt.Printf("%+v\n",o)
}

type Object struct {
   Name string
   Time time.Duration
}

func (o *Object) UnmarshalJSON(data []byte) error {
   tmp := struct {
      Name   string `json:"name"`
      Duration string    `json:"duration"`
   }{}
   err := json.Unmarshal(data,&tmp)
   if err != nil {
      return err
   }
   dur,err := time.ParseDuration(tmp.Duration)
   if err != nil {
      return err
   }
   o.Name = tmp.Name
   o.Time = dur
   return nil
}

	tmp := struct {

		fmt.Printf("failed to parse duration: %v", err)

   "encoding/json"
   "fmt"
   "time"
)

type Nested struct {
   Dur time.Duration `json:"duration"`
}

func (obj *Object) UnmarshalJSON(data []byte) error {
   tmp := struct {
      Dur string `json:"duration"`
      Num int    `json:"num"`
   }{}

      return err

   dur, err := time.ParseDuration(tmp.Dur)

      return err

type Object struct {
   Nested
   Num int `json:"num"`
}

var _ json.Unmarshaler = (*Object)(nil)

   obj := Object{}
   _ = json.Unmarshal([]byte(testJSON), &obj)
   fmt.Printf("result: %+v \n", obj)
}
